# git.txt - QMOI Git/GitHub/Hugging Face Automation Reference

## Required Git Commands for Automation

# Standard push to GitHub

git add .
git commit -m "Automated update by QMOI"
git push origin main

# Push to Hugging Face (using git-lfs if needed)

git remote add hf https://huggingface.co/<username>/<repo>
git push hf main

# Auto-create repo if missing (GitHub CLI example)

gh repo create <username>/<repo> --public --source=. --remote=origin --push

# Auto-create repo if missing (Hugging Face CLI example)
huggingface-cli repo create <repo> --type=model --organization <org>
