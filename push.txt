# Auto Git Push Sequence (with retries)
git add .
git commit -m "WIP: Auto-save unstaged changes before rebase" || echo "No changes to commit."
git pull origin main --rebase || (echo "Pull failed, retrying..." && git pull origin main --rebase)
git push origin main || (echo "Push failed, retrying..." && git push origin main)
